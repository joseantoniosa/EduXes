 49 function open_daily_page() {
 50     var this_date = $("#daily_date").val();
 51     if (this_date != "") {
 52         var a_date = new Date(this_date);
 53         if (a_date.getDay() == 6 || a_date.getDay() == 0) {
 54             help("weekend");
 55         } else {
 56             $.mobile.showPageLoadingMsg();
 57             global_week_day = a_date.getDay();  // 1=> Monday
 58             var week_day = clone(global_week_day);
 59             loadSchedule(global_db, week_day);
 60             global_actual_date = new Date(this_date);
 61             // Reports date is, in origin, the same as actual_date
 62             global_reports_date = global_actual_date ;
 63             $("#current_day").text(this_date);
 64             $.mobile.changePage("#daily_schedule");
 65         }
 66     }

185  * called in: queryStudentsAttendanceSuccess

187 function studentState(id_student, id_group, id_session) {
188     var state =$("#select_student_"+id_student+" option:selected").text()   ;
189     var real_state=STATE_NONE;
190 
191     switch(state) {
192         case 'Absence':
193             real_state=STATE_ABSENCE;
194             break;
195         case 'Unpunctuality':
196             real_state=STATE_UNPUNCTUAL;
197             break;
198         case 'Excused':
199             real_state=STATE_EXCUSED;
200             break;
201         case 'Behavior':
202             real_state=STATE_BEHAVIOR;
203             break;
204         default:
205             real_state=STATE_NONE;
206             break;
207     }
208     // Consider Today as default date.
209     if(global_actual_date!=null) {
210         var actual_date = global_actual_date;
211     } else  if($('#daily_date').val() != null ) {
212         var actual_date = $('#daily_date').val();
213     } else {
214         var actual_date = new Date();
215     }
216     updateStudentState( global_db, id_student, id_group, id_session, real_state, actual_date );
217 }

242 function deleteRecord(db, table, id){
243        var db2= db;
244        var id2= id;
245        var table2 = table;
246        deleteRawRecord(db2, table2, id2);
247        switch(table2) {
248        case 'groups':
249            loadGroups(db2);
250            break;
251        case 'students':
252            loadStudents(db2);
253            break;
254        case 'activities':
255            loadActivities(db2);
256            break;
257        case 'attendance':
258            loadGroups(db2);
259            break;
260        default:
261            break;
262        }
263 }

266 function help(field) {
267     var message;
268     var title;
269     switch (field) {
270         case "student_attendance":
271             title = "student_attendance";
272             message ="Student Attendance ...";
273             break;
274         case "date":
275             title = "Date format";
276             message =" MM/DD/YY";
277             break;
278         case "weekend":
279             title = "Classes";
280             message =" No class on Weekend! ";
281             break;
282         case "no_data":
283             title = "No Data";
284             message =" There is no data available ";
285             break;
286         default:
287             break;
288         }
289         navigator.notification.alert(
290             message,
291             null,
292             "Ayuda: " + title,
293             "Cerrar"
294         );
295 }

305 function textDate(a_today){
306     var today =  (a_today.getMonth() +1 )+ "/" + a_today.getDate() +"/"+  (1900+a_today.getYear());
307     return today;
308 }
309 function textDateYear(a_today){
310     var today =  (a_today.getMonth() +1 )+ "/" + a_today.getDate() ;
311     return today;

313 function clone(obj) {
314     if (null == obj || "object" != typeof obj) return obj;
315     var copy = obj.constructor();
316     for (var attr in obj) {
317         if (obj.hasOwnProperty(attr)) copy[attr] = obj[attr];
318     }
319     return copy;
320 }
